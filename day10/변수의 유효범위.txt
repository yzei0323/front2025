

자바스크립트 변수와 유효범위

선언된 위치에 따라서 다음과 같이 분류할 수 있다

전역변수 : 함수 밖에서 선언  : 모든 함수에서 사용 가능  , 웹페이지가 닫히거나 리로드될 때 까지
지역변수 : 함수 안에서 선언  : 함수안에서만 사용가능, 함수는 반환시  모든 변수는 정리됨 
블럭변수:  블럭(for, if  등 {   } 중괄호 블럭안에서 선언된 변수 :   특정 블럭안에서만 사용가능  , 블럭은 벗어나면 정리됨




//블럭변수 사용예시 
for( let i=1; i<=5; i++){      // i는 for문 안에서만 유효함 , 블럭이 끝나면 정리됨
    console.log( " hi ");
}

for( let i=1; i<=5; i++){  // i는  for문 안에서만 유효만, 블럭이 끝나면 정리됨
    console.log( "bye" );
}


function hello(){
    let cnt=5;              // cnt변수는 함수가 호출되면 만들어졌다가 함수 반환시 정리됨
    console.log( cnt );
}

function hello2(){       // cnt변수는 함수가 호출되면 만들어졌다가 함수 반환시 정리됨
   let cnt=4;
   console.log(  cnt);
}
